\author{Eldad Haber\thanks{Department of Earth and Ocean Science, The University of British Columbia, Vancouver, BC, Canada}}

\maketitle

\begin{enumerate}
\item Differentiate the following expressions
\begin{enumerate}
\item $y = x^n$ \\  \\ \\
\item $y = {\frac {\log(x)}{x}}$ \\   \\   \\
\item Compute ${\frac {\partial z}{\partial x}}$ where $z = x \sin(y)$ \\   \\   \\
\end{enumerate}
\item Integrate the following expressions
\begin{enumerate}
\item $y = {\frac 1x}$ \\  \\ \\
\item $y = x \sin(x)$ \\  \\ \\
\end{enumerate}
\item Use first order Taylor's expansion to compute 
$\cos(0.01)$ (recall that $\cos(0)=1$)  \\  \\ \\ \\

\item Compute the dot product of the vectors
${\bf a} = [1,2,1]$ and ${\bf b} = [1,2,3]$. \\  \\ \\ \\

\item Compute the product of the matrix ${\bf A}$ and the matrix ${\bf x}$ where
$$ {\bf A} = \begin{pmatrix} 0 & 1 & -1 \\ -1 & 0  & 2 \\ 1 & -2 & 0 \end{pmatrix} \quad \quad 
{\bf x} = \begin{pmatrix} 1 \\ 2 \\ 3 \end{pmatrix} $$  \\  \\ \\ \\

\item Solve the linear system
$$  \begin{pmatrix} 2 & 1  \\ 1 & 2   \end{pmatrix}   \begin{pmatrix} x_1 \\x_ 2 \end{pmatrix} =
 \begin{pmatrix} 4 \\ 5 \end{pmatrix} $$  \\  \\ \\ \\

\item Find the eigenvalues of the matrix
$$  \begin{pmatrix} 2 & 1  \\ 1 & 2   \end{pmatrix}    $$  \\  \\ \\ \\

 
\end{enumerate}
